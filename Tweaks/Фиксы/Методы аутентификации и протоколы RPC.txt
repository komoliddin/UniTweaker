ROOT=HKLM
KEY=SOFTWARE\Policies\Microsoft\Windows NT\Printers\RPC
NAME=RpcAuthentication
TYPE=REG_DWORD

Reg.exe add "HKLM\SOFTWARE\Policies\Microsoft\Windows NT\Printers\RPC" /v "RpcUseNamedPipeProtocol" /t REG_DWORD /d "1" /f

Reg.exe add "HKLM\SOFTWARE\Policies\Microsoft\Windows NT\Printers\RPC" /v "RpcOverNamedPipes" /t REG_DWORD /d "1" /f
