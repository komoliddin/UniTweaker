CHECKBOX:
ROOT=HKLM
KEY=SOFTWARE\Microsoft\Windows\Dwm
NAME=EnableHWAcceleration
TYPE=REG_DWORD
VALUE=1

CHECKED:
Reg.exe add "HKLM\SOFTWARE\Microsoft\Windows\Dwm" /v "EnableHWAcceleration" /t REG_DWORD /d "1" /f

UNCHECKED:
Reg.exe add "HKLM\SOFTWARE\Microsoft\Windows\Dwm" /v "EnableHWAcceleration" /t REG_DWORD /d "0" /f
