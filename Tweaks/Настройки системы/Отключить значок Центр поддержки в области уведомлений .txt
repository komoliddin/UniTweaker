CHECKBOX:
ROOT=HKCU
KEY=Software\Microsoft\Windows\CurrentVersion\Policies\Explorer
NAME=HideSCAHealth
TYPE=REG_DWORD
VALUE=1

CHECKED:
Reg.exe add "HKCU\Software\Microsoft\Windows\CurrentVersion\Policies\Explorer" /v "HideSCAHealth" /t REG_DWORD /d "1" /f

UNCHECKED:
Reg.exe add "HKCU\Software\Microsoft\Windows\CurrentVersion\Policies\Explorer" /v "HideSCAHealth" /t REG_DWORD /d "0" /f
