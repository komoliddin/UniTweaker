CHECKBOX:

KEY=Control Panel\Desktop
NAME=CaretBlinkTime
TYPE=REG_SZ
VALUE="250"

CHECKED:
Reg.exe add "HKCU\Control Panel\Desktop" /v "CaretBlinkTime" /t REG_SZ /d "250" /f

UNCHECKED:
Reg.exe add "HKCU\Control Panel\Desktop" /v "CaretBlinkTime" /t REG_SZ /d "530" /f
